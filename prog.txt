# This simple program takes the
# first two characters by ASCII code
# from user input, then returns them in reverse

49 00 0D  # Store char at memory address 0D
49 00 0E  # Store char at memory address 0E
51 00 0E  # Print contents of memory address 0E
51 00 0D  # Print contents of memory address 0D
00        # Terminate
